`id` int(11) NOT NULL AUTO_INCREMENT,
`compared_type` int(11) DEFAULT '0' COMMENT '对比类型(默认0:资源,1:试题)',
`resourse_id` int(11) DEFAULT NULL COMMENT '资源id(对比类型:0,存hxy_resource.id。对比类型:1,存questions_pk.id)',

`create_time` datetime(0) NOT NULL DEFAULT CURRENT_TIMESTAMP(0) COMMENT '创建时间',
`update_time` datetime(0) NULL DEFAULT CURRENT_TIMESTAMP(0) COMMENT '更新时间，自动修改'
PRIMARY KEY (`id`),
KEY `resourse_id` (`resourse_id`) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 4 CHARACTER SET = utf8mb4 COLLATE = utf8mb4_general_ci ROW_FORMAT = Dynamic;
